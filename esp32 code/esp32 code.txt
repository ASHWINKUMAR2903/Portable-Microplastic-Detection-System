/* -------------------------------------------------------------
   Microplastic Detection System (Professional IoT Version)
   Hardware: ESP32 + ADS1115 + BPW34
   Backend: Node.js (Express) on Render via HTTPS POST

   Design:
   - ESP32 samples fast
   - Sends summarized data per time window
   - Deterministic (NO ML)
   - Posts to Render cloud backend via HTTPS
-------------------------------------------------------------- */

#include <Wire.h>
#include <Adafruit_ADS1X15.h>
#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <HTTPClient.h>
#include <time.h>

// ================== OBJECTS ==================
Adafruit_ADS1115 ads;

// ================== WIFI ==================
const char* WIFI_SSID = "ASHWIN";
const char* WIFI_PASS = "12345678";

// ================== BACKEND (RENDER HTTPS URL) ==================
const char* SERVER_URL = "https://portable-microplastic-detection-system-1.onrender.com/api/telemetry";

// ================== NTP TIME CONFIG ==================
const char* NTP_SERVER = "pool.ntp.org";
const long  GMT_OFFSET = 19800;     // IST = UTC+5:30 = 19800 seconds
const int   DAYLIGHT_OFFSET = 0;

// ================== TIMING ==================
const int READ_INTERVAL_MS   = 100;     // sensor sampling
const int WINDOW_MS          = 10000;   // reporting window
const int POST_RETRY_DELAY   = 3000;
const int MAX_POST_RETRIES   = 3;

// ================== SIGNAL ==================
const int   MOVING_AVG_N = 10;
const float EVENT_THRESHOLD_MULT = 1.5;

// ================== GLOBALS ==================
unsigned long lastRead       = 0;
unsigned long windowStart    = 0;
unsigned long lastPostTry    = 0;

float samples[MOVING_AVG_N];
int sampleIndex = 0;

float recentAvg   = 0.0;
float baseline    = 0.0;
float baselineStd = 0.0;
bool  baselineReady = false;

unsigned long lastEventTime = 0;
int eventCountWindow = 0;
int eventCountTotal  = 0;

// ================== UTILS ==================
float adcToVoltage(int16_t raw) {
  return raw * (4.096 / 32768.0);
}

// ================== WIFI ==================
void ensureWiFi() {
  if (WiFi.status() == WL_CONNECTED) return;

  Serial.println("WiFi disconnected, reconnecting...");
  WiFi.disconnect();
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  unsigned long start = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - start < 10000) {
    delay(500);
    Serial.print(".");
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nWiFi reconnected");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\nWiFi reconnect failed");
  }
}

// ================== TIME ==================
long getEpoch() {
  time_t now;
  time(&now);
  if (now < 100000) return millis() / 1000;  // fallback before NTP sync
  return now;
}

// ================== BASELINE ==================
void computeBaseline() {
  const int N = 200;
  double sum = 0, sumsq = 0;

  Serial.println("Calibrating baseline...");
  for (int i = 0; i < N; i++) {
    int16_t raw = ads.readADC_SingleEnded(0);
    float v = -adcToVoltage(raw);
    sum += v;
    sumsq += v * v;
    delay(10);
  }

  baseline = sum / N;
  double var = (sumsq / N) - (baseline * baseline);
  baselineStd = var > 0 ? sqrt(var) : 0;
  baselineReady = true;

  Serial.print("Baseline = "); Serial.println(baseline, 6);
  Serial.print("Noise STD = "); Serial.println(baselineStd, 6);
}

// ================== HTTPS POST ==================
void sendWindowToServer(float avgVoltage) {
  ensureWiFi();
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("No WiFi — skipping send");
    return;
  }

  // Build JSON payload
  String payload = "{";
  payload += "\"device_id\":\"esp32-001\",";
  payload += "\"timestamp\":" + String(getEpoch()) + ",";
  payload += "\"window_ms\":" + String(WINDOW_MS) + ",";
  payload += "\"light_voltage\":" + String(avgVoltage, 6) + ",";
  payload += "\"event_count_recent\":" + String(eventCountWindow) + ",";
  payload += "\"baseline\":" + String(baseline, 6) + ",";
  payload += "\"noise_std\":" + String(baselineStd, 6) + ",";
  payload += "\"event_count\":" + String(eventCountTotal) + ",";
  payload += "\"event_detected\":" + String(eventCountTotal > 0 ? "true" : "false");
  payload += "}";

  Serial.println("Sending: " + payload);

  // Retry loop for reliability
  for (int attempt = 1; attempt <= MAX_POST_RETRIES; attempt++) {
    WiFiClientSecure client;
    client.setInsecure();  // Skip certificate verification (OK for prototype)

    HTTPClient http;
    http.begin(client, SERVER_URL);
    http.addHeader("Content-Type", "application/json");
    http.setTimeout(10000);  // 10 second timeout

    int code = http.POST(payload);
    String response = http.getString();
    http.end();

    if (code == 200 || code == 201) {
      Serial.print("✅ Data sent (attempt ");
      Serial.print(attempt);
      Serial.print("): HTTP ");
      Serial.println(code);
      Serial.println("Response: " + response);
      return;  // Success — exit retry loop
    } else {
      Serial.print("❌ Attempt ");
      Serial.print(attempt);
      Serial.print(" failed: HTTP ");
      Serial.println(code);
      if (code < 0) {
        Serial.println("Connection error (check WiFi/server)");
      }
      if (attempt < MAX_POST_RETRIES) {
        Serial.println("Retrying in " + String(POST_RETRY_DELAY) + "ms...");
        delay(POST_RETRY_DELAY);
      }
    }
  }

  Serial.println("❌ All retries exhausted — data lost for this window");
}

// ================== SETUP ==================
void setup() {
  Serial.begin(115200);
  delay(200);

  Serial.println("\n========================================");
  Serial.println("  Microplastic Detector v2.0 Starting");
  Serial.println("========================================");

  // ADS1115 init
  if (!ads.begin()) {
    Serial.println("ERROR: ADS1115 NOT FOUND!");
    while (1);
  }
  ads.setGain(GAIN_ONE);
  Serial.println("ADS1115 initialized");

  // WiFi connect
  WiFi.mode(WIFI_STA);
  WiFi.setSleep(false);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  Serial.print("Connecting WiFi");
  unsigned long wifiStart = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - wifiStart < 15000) {
    delay(500);
    Serial.print(".");
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nWiFi Connected");
    Serial.print("IP: "); Serial.println(WiFi.localIP());
  } else {
    Serial.println("\nWiFi FAILED — will retry in loop");
  }

  // NTP Time Sync
  Serial.println("Syncing time via NTP...");
  configTime(GMT_OFFSET, DAYLIGHT_OFFSET, NTP_SERVER);

  // Wait for time sync (up to 10s)
  int ntpRetries = 0;
  time_t now = 0;
  while (now < 100000 && ntpRetries < 20) {
    delay(500);
    time(&now);
    ntpRetries++;
    Serial.print(".");
  }

  if (now > 100000) {
    struct tm timeinfo;
    localtime_r(&now, &timeinfo);
    Serial.print("\nTime synced: ");
    Serial.println(asctime(&timeinfo));
  } else {
    Serial.println("\nNTP sync failed — using millis() fallback");
  }

  // Init moving average buffer
  float init = -adcToVoltage(ads.readADC_SingleEnded(0));
  for (int i = 0; i < MOVING_AVG_N; i++) samples[i] = init;

  windowStart = millis();

  Serial.println("========================================");
  Serial.println("  System Ready. Press 'c' to calibrate");
  Serial.println("========================================");
  Serial.print("Backend URL: "); Serial.println(SERVER_URL);
}

// ================== LOOP ==================
void loop() {
  if (Serial.available()) {
    char c = Serial.read();
    if (c == 'c') computeBaseline();
  }

  unsigned long now = millis();
  if (now - lastRead < READ_INTERVAL_MS) return;
  lastRead = now;

  // ----- SENSOR READ -----
  float light = -adcToVoltage(ads.readADC_SingleEnded(0));

  samples[sampleIndex++] = light;
  if (sampleIndex >= MOVING_AVG_N) sampleIndex = 0;

  float sum = 0;
  for (int i = 0; i < MOVING_AVG_N; i++) sum += samples[i];
  recentAvg = sum / MOVING_AVG_N;

  // ----- EVENT DETECTION -----
  if (baselineReady) {
    float threshold = baseline + EVENT_THRESHOLD_MULT * baselineStd;
    if (recentAvg > threshold && now - lastEventTime > 150) {
      eventCountWindow++;
      eventCountTotal++;
      lastEventTime = now;
    }

    Serial.print("V=");
    Serial.print(recentAvg, 6);
    Serial.print(" Th=");
    Serial.print(threshold, 6);
    Serial.print(" WEvents=");
    Serial.println(eventCountWindow);
  }

  // ----- WINDOW SEND -----
  if (now - windowStart >= WINDOW_MS) {
    sendWindowToServer(recentAvg);
    eventCountWindow = 0;
    windowStart = now;
  }
}
